class Solution {
public:
    string removeOccurrences(string s, string part) {
        // s = "daabcbaabcbc"
        string result = "";
        int n = s.length();
        int m = part.length();
        stack<char> st;
        
        for(char &ch: s) {
            result.push_back(ch);
            //if result lenght(5) >= part length(3) && in ababc substr(5 - 3) that is abc == part(abc), true
            if(result.length() >= m && result.substr(result.length() - m) == part) {
                for(int i = 0; i < m; i++) { //the commented part and this line works same
                    result.pop_back();
                }
                // result.erase(result.length() - m); 
            }
        }

        return result;
    }
};
